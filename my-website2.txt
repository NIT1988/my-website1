<body>
    <div class="container">
        <h1>Магазин Хендмейд</h1>
        <div id="products">
            <!-- Products will be added dynamically here -->
        </div>
        <div class="cart">
            <h3>Кошик</h3>
            <ul id="cart-items"></ul>
            <p><strong>Загальна сума:</strong> <span id="total-price">0</span> грн</p>
            <button id="checkout" class="checkout" disabled>Перейти до оплати</button>
        </div>
    </div>
    <script>
        const products = Array.from({ length: 50 }, (_, i) => ({
            id: i + 1,
            name: `Товар ${i + 1}`,
            price: Math.floor(Math.random() * 1000) + 100,
            img: "https://via.placeholder.com/100"
        }));

        const cart = [];

        // Display products
        const productsContainer = document.getElementById("products");
        products.forEach(product => {
            const productDiv = document.createElement("div");
            productDiv.className = "product";
            productDiv.innerHTML = `
                <img src="${product.img}" alt="${product.name}">
                <div>
                    <h2>${product.name}</h2>
                    <p>Ціна: ${product.price} грн</p>
                    <button onclick="addToCart(${product.id})">Додати в кошик</button>
                </div>
            `;
            productsContainer.appendChild(productDiv);
        });

        // Add to cart
        function addToCart(id) {
            const product = products.find(p => p.id === id);
            cart.push(product);
            updateCart();
        }

        // Update cart
        function updateCart() {
            const cartItems = document.getElementById("cart-items");
            const totalPrice = document.getElementById("total-price");
            const checkoutButton = document.getElementById("checkout");

            cartItems.innerHTML = "";
            let total = 0;

            cart.forEach(item => {
                const li = document.createElement("li");
                li.textContent = `${item.name} - ${item.price} грн`;
                cartItems.appendChild(li);
                total += item.price;
            });

            totalPrice.textContent = total;
            checkoutButton.disabled = cart.length === 0;
        }

        // Checkout functionality
        document.getElementById("checkout").addEventListener("click", () => {
            alert("Дякуємо за покупку! Ваше замовлення прийняте.");
            cart.length = 0;
            updateCart();
        });
    </script>
